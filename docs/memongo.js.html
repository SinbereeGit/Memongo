/**
 * Memongo
 *
 * @summary A minimal MongoDB-like in-memory database with persistence capabilities through user-provided read/write functions.
 *
 * @tutorial https://github.com/SinbereeGit/Memongo
 * @author Sinber (https://github.com/SinbereeGit)
 * @license GNU GPL v3 (https://www.gnu.org/licenses/gpl-3.0.html)
 *
 * @description
 * Provides a MongoDB-like API for manipulating JSON data stored in memory with persistence capabilities through user-provided read/write functions.
 *
 * @note
 * Each write operation rewrites the entire database, which may lead to poor
 * performance for large datasets.
 *
 * @warning
 * If an exception occurs or is ignored during an operation, you must call `init()` again before using it again, or something bad and unpredicatable may happen.
 */

/**
 * Error message constants used by *Memongo*.
 * These represent different error types that may occur during operations in *Memongo*.
 *
 * @readonly
 */
const LOCALDB_ERROR_TYPES = Object.freeze({
  LOCALDB_NOT_INITIALIZED: "Local database not initialized",
  COLLECTION_ALREADY_EXISTS: "Collection already exists",
  COLLECTION_NOT_EXISTS: "Collection not exists",

  INVALID_PARAMETER_TYPE: "Invalid parameter type",
  SORT_INCOMPARABLE: "Incomparable value during sorting",

  DOCUMENT_ALREADY_EXISTS: "Document already exists",
  DOCUMENT_NOT_EXISTS: "Document not exists",
});

/**
 * Utility class for manipulating plain JSON-like objects using dot-notation paths.
 *
 * Provides helper methods for:
 * - Retrieving nested values
 * - Setting nested values
 * - Performing partial updates
 * - Cloning JSON objects
 *
 * All path-based operations support **dot notation** (e.g. `"user.profile.name"`).
 *
 * @example
 * const obj = {
 *   user: {
 *     profile: {
 *       age: 20
 *     }
 *   }
 * };
 *
 * JSONObjUtils.getByPath(obj, "user.profile.age"); // 20
 *
 * JSONObjUtils.setByPath(obj, "user.profile.age", 21);
 *
 * JSONObjUtils.update(obj, {
 *   "user.profile.age": (age) => age + 1
 * });
 */
class _JSONObjUtils {
  /**
   * Retrieves a value from a nested JSON object using a dot-notation path
   *
   * @param {Object} jsonObj - The JSON object to traverse
   * @param {string} path - Dot-notation string representing the path to the desired value
   *
   * @returns {*} Returns the value at the specified path, or `undefined` if:
   *          - The path doesn't exist
   *          - Any intermediate property is `null` or `undefined`
   */
  static getByPath = (jsonObj, path) => {
    const keys = path.split(".");
    let cur = jsonObj;

    for (let k of keys) {
      if (cur == null) return undefined;
      cur = cur[k];
    }

    return cur;
  };

  /**
   * Sets a value in a nested JSON object using a dot-notation path, creating intermediate objects if needed
   *
   * @param {Object} jsonObj - The JSON object to modify
   * @param {string} path - Dot-notation string representing the path where the value should be set
   * @param {*} value - The value to set at the specified path
   */
  static setByPath = (jsonObj, path, value) => {
    const keys = path.split(".");
    let cur = jsonObj;

    for (let i = 0; i < keys.length - 1; i++) {
      const k = keys[i];

      if (typeof cur[k] !== "object" || cur[k] === null) {
        cur[k] = {};
      }

      cur = cur[k];
    }

    cur[keys[keys.length - 1]] = value;
  };

  /**
   * Creates a deep clone of an JSON object using JSON serialization
   *
   * @param {Object} jsonObj - The JSON object to clone
   *
   * @returns {Object} A deep copy of the input JSON object
   */
  static clone = (jsonObj) => JSON.parse(JSON.stringify(jsonObj));

  /**
   * Updates an JSON object in place using a partial update object
   *
   * @param {Object} jsonObj - The target JSON object to update (modified in place)
   * @param {Object<string, (*|function(*):*)>} update The update specification object where:
   *        - Keys are dot-notation paths to target properties
   *        - Properties with `undefined` values are skipped
   *        - Values can be either:
   *          - A direct value to set at the path
   *          - A function that receives the current value and returns the new value
   *
   * @example
   * updateObject(user, {
   *   "name": "New Name",                    // Direct value
   *   "profile.age": (current) => current + 1, // Functional update
   *   "address.city": undefined               // Skipped
   * });
   */
  static update(jsonObj, update) {
    for (const key in update) {
      if (update[key] === undefined) continue;

      const value =
        typeof update[key] === "function"
          ? update[key](_JSONObjUtils.getByPath(jsonObj, key))
          : update[key];

      _JSONObjUtils.setByPath(jsonObj, key, value);
    }
  }
}

/**
 * Represents a collection of documents in the database, providing methods for querying, manipulating, and managing records
 *
 * @example
 * // Create a new collection instance
 * const users = new Collection(userData, rootDatabase, { orderBy: [{ field: "name", order: "asc" }] });
 *
 * @example
 * // Chain query methods
 * const results = collection
 *   .where({ age: { $gt: 18 } })
 *   .orderBy("name", "asc")
 *   .limit(10)
 *   .skip(5)
 *   .get();
 */
class _Collection {
  /**
   * Creates a new Collection instance
   *
   * @constructor
   * @param {MemoryJSONDB} root - The root of the collection
   * @param {Object} content - The collection content object where keys are document IDs and values are document contents
   * @param {Object} [options] - Configuration options for query operations
   */
  constructor(root, content, options = { orderBy: [] }) {
    this.root = root;
    this.content = content;
    this.options = options;
    if (!this.options.orderBy) this.options.orderBy = [];
  }

  /**
   * Add a record, automatically generating an `_id` if not specified
   *
   * @param {Object} options - Configuration options
   * @param {Object} options.data - The data object to be added
   * @param {string} [options.data._id] - Optional custom _id, automatically generated if not provided
   *
   * @returns {Promise<Object|null>} Returns an object containing the _id of the new record:
   *          - Success: `{ _id: string }`
   *          - Failure: `null`
   *
   * @throws {Error} When the provided parameter is not an object
   * @throws {Error} When the _id in the object already exists
   */
  async add({ data }) {
    const generateRandomID = () => {
      let id = "";
      const map =
        "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";
      while (!id || this.content[id]) {
        for (let i = 0; i < 4; i++) id += map[parseInt(Math.random() * 62)];
      }
      return id;
    };

    if (typeof data !== "object")
      throw new Error(LOCALDB_ERROR_TYPES.INVALID_PARAMETER_TYPE);

    if (data._id && this.content[data._id])
      throw new Error(LOCALDB_ERROR_TYPES.DOCUMENT_ALREADY_EXISTS);

    const clone = _JSONObjUtils.clone(data);
    delete clone._id;

    const id = data._id || generateRandomID();
    this.content[id] = clone;

    await this.root.write();

    return { _id: id };
  }

  /**
   * Get a record
   * @param {String} id - The id of the record to retrieve
   * @returns {_Document} Returns a Document object
   *
   * @throws {Error} When the specified id does not exist
   */
  doc(id) {
    const data = this.content[id];

    if (data === undefined)
      throw new Error(LOCALDB_ERROR_TYPES.DOCUMENT_NOT_EXISTS);

    return new _Document(this, id, data);
  }

  /**
   * Query records
   * @param {Object} conditions - Query conditions (supports nested field paths, commands, regular expressions)
   * @returns Returns the found collection
   */
  where(conditions) {
    const match = (value, condition) => {
      if (condition instanceof _Command) {
        return condition.exec(value);
      }

      if (condition instanceof RegExp) {
        return condition.test(value);
      }

      if (
        condition &&
        typeof condition === "object" &&
        !Array.isArray(condition)
      ) {
        if (typeof value !== "object" || value === null) return false;

        for (const key in condition) {
          if (!match(value[key], condition[key])) {
            return false;
          }
        }

        return true;
      }

      return value === condition;
    };

    const query = (key) => {
      for (const keyInConditions in conditions) {
        const doc =
          keyInConditions === "_id"
            ? key
            : _JSONObjUtils.getByPath(this.content[key], keyInConditions);
        if (!match(doc, conditions[keyInConditions])) return false;
      }
      return true;
    };

    const res = {};

    Object.keys(this.content).forEach(
      (key) => query(key) && (res[key] = this.content[key]),
    );

    return new _Collection(this.root, res, this.options);
  }

  /**
   * Set an upper limit on the number of query results
   * @param {Number} limit - Maximum number of results
   * @returns {_Collection} Returns the collection after setting the limit
   */
  limit(limit) {
    const options = _JSONObjUtils.clone(this.options);
    this.options.limit = limit;
    return new _Collection(this.root, this.content, options);
  }

  /**
   * Skip the specified number of results before returning query results
   * @param {Number} skip - Number of results to skip
   * @returns {_Collection} Returns the collection after setting the skip
   */
  skip(skip) {
    const options = _JSONObjUtils.clone(this.options);
    options.skip = skip;
    return new _Collection(this.root, this.content, options);
  }

  /**
   * Specify query sorting criteria (supports nested fields)
   * @param {String} field - Field to sort by
   * @param {('asc'|'desc')} order - Ascending or descending order
   * @returns {_Collection} Returns the collection after setting the sort order
   */
  orderBy(field, order = "asc") {
    const options = _JSONObjUtils.clone(this.options);
    options.orderBy.push({
      field,
      order,
    });
    return new _Collection(this.root, this.content, options);
  }

  /**
   * Count the number of records that match the query conditions
   * @returns {Object} The `total` property contains the number of records
   */
  count() {
    return { total: Object.keys(this.content).length };
  }

  /**
   * Retrieve collection data
   * @returns {Object} Returns `{data: Object}`, where `data` is an array of query results
   *
   * @throws {Error} When sorting is specified but the sort field contains incomparable values
   */
  get() {
    const res = [];

    const add = (key) =>
      res.push(_Document.getDocWithId(this.content[key], key));

    const compare = (a, b) => {
      if (a === b) return 0;

      if (typeof a === "string" && typeof b === "string")
        return a.localeCompare(b);

      if (typeof a !== "number" && typeof b !== "number")
        throw new Error(LOCALDB_ERROR_TYPES.SORT_INCOMPARABLE);

      return a > b ? 1 : -1;
    };

    Object.keys(this.content).forEach((key) => add(key));

    if (this.options.orderBy?.length)
      res.sort((a, b) => {
        let result = 0;
        for (let i = 0; i < this.options.orderBy.length; i++) {
          const item = this.options.orderBy[i];
          const valueOfA = _JSONObjUtils.getByPath(a, item.field),
            valueOfB = _JSONObjUtils.getByPath(b, item.field);
          result = compare(valueOfA, valueOfB);
          if (!result) continue;
          return (item.order == "desc" ? -1 : 1) * result;
        }
        return 0;
      });

    const { skip = 0, limit } = this.options;
    res.splice(0, skip);
    if (limit != null) res.splice(limit);

    return { data: res };
  }

  /**
   * Update multiple records (supports nested path updates)
   * @param {Object} options - Configuration options
   * @param {Object} options.data - The content to update
   *
   * @throws {Error} When the provided parameter is not an object
   */
  async update({ data }) {
    if (typeof data !== "object")
      throw new Error(LOCALDB_ERROR_TYPES.INVALID_PARAMETER_TYPE);

    Object.keys(this.content).forEach((key) =>
      _JSONObjUtils.update(this.content[key], data),
    );

    await this.root.write();
  }

  /**
   * Delete all documents in the collection
   */
  async remove() {
    Object.keys(this.content).forEach((key) => delete this.content[key]);

    await this.root.write();
  }

  async removeById(id) {
    if (this.content[id] === undefined)
      throw new Error(LOCALDB_ERROR_TYPES.DOCUMENT_NOT_EXISTS);

    delete this.content[id];
  }

  async write() {
    await this.root.write();
  }
}

/**
 * Represents a document in the `Collection`.
 *
 * A `Document` instance provides methods to read, update, and remove
 * a single document stored inside a `Collection` as its root.
 *
 * @example
 * const doc = new Document(collectionAsRoot, "user123", { name: "Tom", age: 20 });
 *
 * const result = doc.get();
 * console.log(result.data._id); // "user123"
 *
 * await doc.update({
 *   data: {
 *     "age": (age) => age + 1
 *   }
 * });
 *
 * doc.remove();
 */
class _Document {
  /**
   * Creates a new Document instance representing a single record in the collection
   *
   * @constructor
   * @param {_Collection} root - The root of the document
   * @param {string} id - The unique identifier of this document
   * @param {Object} content - The document's data content
   */
  constructor(root, id, content) {
    this.root = root;
    this.id = id;
    this.content = content;
  }

  /**
   * Creates a document object with an attached _id property
   *
   * @static
   * @param {Object} doc - The original document object without an `"_id"`
   * @param {string} id - The identifier to attach to the document
   *
   * @returns {Object} A new object containing all properties of the original document plus an `"_id"` field
   */
  static getDocWithId = (doc, id) => {
    const docWithId = _JSONObjUtils.clone(doc);
    docWithId._id = id;
    return docWithId;
  };

  /**
   * Get document data
   * @returns {Object} Returns `{data: Object}`, where the `data` field contains the document
   */
  get() {
    return { data: _Document.getDocWithId(this.content, this.id) };
  }

  /**
   * Update a document
   * @param {Object} options
   * @param {Object} options.data - Update content
   *
   * @throws {Error} When the update content is not an object
   */
  async update({ data }) {
    if (typeof data !== "object")
      throw new Error(LOCALDB_ERROR_TYPES.INVALID_PARAMETER_TYPE);

    _JSONObjUtils.update(this.content, data);

    await this.root.write();
  }

  /**
   * Delete this document
   */
  remove() {
    this.root.removeById(this.id);
  }
}

/**
 * Represents a command used to evaluate a condition.
 *
 * @example
 * const isPositive = new Command(v => v > 0);
 * isPositive.exec(5); // true
 */
class _Command {
  /**
   * Creates a new Command instance.
   *
   * @param {(value: any) => boolean} func - Evaluation function.
   * The function receives a value to test and must return `true` or `false`.
   */
  constructor(func) {
    this.exec = func;
  }

  /**
   * Logical AND operation
   * @param {_Command} command - The next query command
   * @returns {_Command} AND query command
   */
  and(cmd) {
    return new _Command((val) => this.exec(val) && cmd.exec(val));
  }

  /**
   * Logical OR operation
   * @param {_Command} command - The next query command
   * @returns {_Command} OR query command
   */
  or(cmd) {
    return new _Command((val) => this.exec(val) || cmd.exec(val));
  }
}

/**
 * A lightweight in-memory JSON database with MongoDB-like style API.
 *
 * The database keeps all data in memory and persists it through user-provided
 * read/write functions.
 *
 * @class
 *
 * @example
 * // In Wechat Miniprogram, using local storage as the persistant method
 * const db = new MemoryJSONDB(
 *   async () => {
 *     try {
 *       return (await wx.getStorage({ key: "localDB" })).data;
 *     } catch {
 *       return {};
 *     }
 *   },
 *   async (jsonObj, resolve, reject) => {
 *     wx.setStorage({
 *       key: "localDB",
 *       data: jsonObj,
 *       success: resolve,
 *       fail: reject,
 *     });
 *   },
 * );
 * await db.init();
 *
 * const users = await db.createCollection("users");
 *
 * @example
 * // The persistant storage stores the JSON string
 * const db = new MemoryJSONDB(
 *   async () => JSON.parse(localStorage.getItem("db")),
 *   async (data, resolve, reject) => {
 *     try {
 *       localStorage.setItem("db", JSON.stringify(data));
 *       resolve();
 *     } catch (err) {
 *       reject(err);
 *     }
 *   }
 * );
 */
class MemoryJSONDB {
  _memoryJSONDB = null;
  _dirty = false;

  /**
   * Create a MemoryJSONDB instance.
   *
   * @param {() => Promise<Object>} getJSONObj Function that returns the persisted JSON object.
   *
   * @param {(jsonObj, resolve, reject) => Promise<void>} writeJSONObj Function that writes the JSON object to persistent storage.
   */
  constructor(getJSONObj, writeJSONObj) {
    this._getJSONObj = getJSONObj;
    this._writeJSONObj = writeJSONObj;
  }

  /**
   * Collection of static command factories for building query conditions
   *
   * @description
   * Provides a comprehensive set of command factories for building complex queries.
   * All commands return Command objects that can be combined using and()/or() methods
   * for advanced query composition.
   */
  static command = {
    // query command (used in `where`)
    eq: (query) => new _Command((val) => val == query),
    neq: (query) => new _Command((val) => val != query),
    lt: (query) => new _Command((val) => val < query),
    lte: (query) => new _Command((val) => val <= query),
    gt: (query) => new _Command((val) => val > query),
    gte: (query) => new _Command((val) => val >= query),
    in: (query) => new _Command((val) => query.includes(val)),
    nin: (query) => new _Command((val) => !query.includes(val)),
    exists: (query) => new _Command((val) => (val == void 0 ? !query : query)),
    and: (...cmds) => new _Command((val) => cmds.every((cmd) => cmd.exec(val))),
    or: (...cmds) => new _Command((val) => cmds.some((cmd) => cmd.exec(val))),

    // TODO: the followings are not supported yet
    // update command (used in `update`)
    // set: (val) => cloneJSONObj(val),
    // remove: () => void 0,
    // inc: (diff) => (val) => (val += diff),
    // mul: (diff) => (val) => (val *= diff),
    // push: (diff) => (val) => (val.push(diff), val),
    // pop: () => (val) => (val.pop(), val),
    // shift: () => (val) => (val.shift(), val),
    // unshift: (diff) => (val) => (val.unshift(diff), val),
  };

  /**
   * Initialize the database
   */
  async init() {
    await this.read();
  }

  /**
   * Writes the in-memory database to persistent storage with debouncing
   *
   * @returns {Promise<void>} A promise that resolves when the write operation completes:
   *          - Resolves immediately if no changes were made during the debounce period
   *          - Resolves after successful storage write
   *          - Rejects if the storage write fails
   *
   * @throws {Error} When the storage write operation fails (e.g., storage quota exceeded)
   *
   * @description
   * This method implements write debouncing by:
   * 1. Marking the database as dirty (needs write)
   * 2. Waiting 200ms for potential additional changes
   * 3. Writing to persistent storage only if still dirty
   *
   * @example
   * // Write with debouncing
   * await db.write();
   * console.log("Database persisted");
   *
   * @example
   * // Handle write errors
   * try {
   *   await db.write();
   * } catch (error) {
   *   console.error("Failed to save database:", error);
   * }
   */
  write() {
    this._dirty = true;

    return new Promise((resolve, reject) => {
      setTimeout(async () => {
        if (!this._dirty) return resolve();

        this._dirty = false;

        await this._writeJSONObj(this._memoryJSONDB, resolve, reject);
      }, 200);
    });
  }

  /**
   * Reads data from persistent storage into memory if not already loaded
   */
  async read() {
    if (!this._memoryJSONDB) {
      this._memoryJSONDB = await this._getJSONObj();
    }
  }

  /**
   * Create a collection
   * @param {String} name - The name of the collection to create
   * @returns {Promise<_Collection>} Returns the collection object
   *
   * @throws {Error} When the local database is not initialized
   * @throws {Error} When the collection already exists
   */
  async createCollection(name) {
    if (!this._memoryJSONDB)
      throw new Error(LOCALDB_ERROR_TYPES.LOCALDB_NOT_INITIALIZED);

    if (this._memoryJSONDB[name])
      throw new Error(LOCALDB_ERROR_TYPES.COLLECTION_ALREADY_EXISTS);

    this._memoryJSONDB[name] = {};

    await this.write();

    return new _Collection(this, this._memoryJSONDB[name]);
  }

  /**
   * Remove a collection
   * @param {String} name - The name of the collection to remove
   *
   * @throws {Error} When the local database is not initialized
   * @throws {Error} When the collection does not exist
   */
  async removeCollection(name) {
    if (!this._memoryJSONDB)
      throw new Error(LOCALDB_ERROR_TYPES.LOCALDB_NOT_INITIALIZED);

    if (!this._memoryJSONDB[name])
      throw new Error(LOCALDB_ERROR_TYPES.COLLECTION_NOT_EXISTS);

    delete this._memoryJSONDB[name];

    await this.write();
  }

  /**
   * Get a collection object
   * @param {String} name - The name of the collection
   * @returns {_Collection | null} Returns the collection object if it exists, otherwise returns `null`
   *
   * @throws {Error} When the localDB is not initialized
   */
  collection(name) {
    if (!this._memoryJSONDB)
      throw new Error(LOCALDB_ERROR_TYPES.LOCALDB_NOT_INITIALIZED);

    return this._memoryJSONDB[name]
      ? new _Collection(this, this._memoryJSONDB[name])
      : null;
  }
}

module.exports = {
  LOCALDB_ERROR_TYPES,
  MemoryJSONDB,
};
